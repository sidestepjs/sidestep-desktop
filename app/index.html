CREATE PROCEDURE [dbo].[A_SP_FETCH_STOCK_MOVEMENT_SUMMARY] (
  @companyCode nvarchar(50)
)
AS
BEGIN
  SET NOCOUNT ON;
  declare @refund int, @wastage int, @writeoff int, @returnstock int;

  set @refund      = ( select sum (quantity) from tbCustomerRefund where companyCode = @companyCode )
  set @wastage     = ( select sum (quantity) from tbWastage where companyCode = @companyCode)
  set @writeoff    = ( select sum (quantity) from tbWriteOff where companyCode = @companyCode)
  set @returnstock = ( select sum (quantity) as returnstock from tbReturnStock where companyCode = @companyCode )

  select @refund as refund, @wastage as wastage, @writeoff as writeoff, @returnstock as returnstock
END
